\section{General relationships}

\begin{align*}
\vb{F}_{\text{net}} &= m\dv[2]{\vb{x}}{t}
&
\vb{F} &= -\grad{U}
&
U &= -\int \vb{F} \text{d}x
\\
K &= \frac{p^2}{2m}
&
a_\phi &= v^2/r
&
W &= -\Delta U
\\
\omega &= 2\pi f
&
T &= 1/f
&
k &= 2\pi/\lambda
\end{align*}



\section{Gravitation}

\begin{align*}
\vb{F} &= \frac{Gm_1m_2}{r^2} \ \hat{r}
&
U &= -\frac{Gm_1m_2}{r}
\\
v &= \sqrt{2GM/r}
&
T^2 &= \frac{4\pi^2}{GM}r^3
\\
\end{align*}




\section{Fluids}


\begin{align*}
P &= \frac{F}{A}
&
P &= P_0 + \rho gd
\\
F_1 &= \frac{A_1}{A_2} F_2 + \rho ghA_1
&
F_B &= \rho_f V_fg
\\
v_1A_1 &= v_2A_2
&
P + 1/2\rho v^2 + \rho gh &= \text{constant}
\end{align*}




\section{Oscillations}

\begin{align*}
x(t) &= A\cos{wt+\phi}
&
\omega &= \sqrt{k/m}
&
\vb{F} &= -k\vb{x}
\\
v_{\text{max}} &= A\omega
&
T &= 2\pi \sqrt{m/k}
&
U &= 1/2kx^2
\end{align*}


\section{Traveling waves}

\begin{align*}
v_{\text{string}} &= \sqrt{T_s/\mu}
&
\mu &= m/L
&
v &= \lambda f
\\
\omega &= vk
&
D(x,t) &= A\cos{(kx + \omega t +\phi)}
&
\pdv[2]{D}{t} &= \frac{T_s}{\mu}\pdv[2]{D}{x}
\end{align*}

\section{Doppler effect}

\begin{align*}
f_+ &= \frac{f_0}{1-v_0/v}
&
\text{approaching source}
\\
f_- &= \frac{f_0}{1+v_0/v}
&
\text{receding source}
\\
f_+ &= (1+v_0/v)f_0
&
\text{approaching observer}
\\
f_- &= (1-v_0/v)f_0
&
\text{receding observer}
\end{align*}
